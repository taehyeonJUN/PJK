    <param-name>spring.profiles.active</param-name>  
    <param-value>dev</param-value>  
</context-param>

    <param-name>spring.profiles.default</param-name>  
    <param-value>dev</param-value>  
</context-param>

    <param-name>spring.liveBeansView.mbeanDomain</param-name>  
    <param-value>dev</param-value>  
</context-param>
